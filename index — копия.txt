function playAudio(audioElement) {
    const playPromise = audioElement.play();
    if (playPromise !== undefined) {
        playPromise.catch(error => {
            console.log("Автовоспроизведение заблокировано:", error);
        });
    }
}

        historyAudio.currentTime = 0;

}

function typeWriter() {
    const element = document.getElementById("historyText");
    let i = 0;
    const speed = 30;
    let isNewLine = false;

    function typing() {
        if (i < fullText.length) {
            if (fullText.charAt(i) === '\n') {
                element.innerHTML += '<br>';
                isNewLine = true;
            } else {
                if (isNewLine && fullText.charAt(i) === ' ') {
                    element.innerHTML += '&nbsp;';
                } else {
                    element.innerHTML += fullText.charAt(i);
                    isNewLine = false;
                }
            }
            
            i++;
            setTimeout(typing, speed);
            element.scrollTop = element.scrollHeight;
        } else {
            element.style.borderRight = "none";
        }
    }
    
    element.innerHTML = '';
    typing();
}

// Функции для работы с модальными окнами
function showModal(modalId) {
    document.getElementById(modalId).style.display = 'flex';
    document.body.style.overflow = 'hidden'; // Запрет прокрутки фона
}

function closeModal(modalId) {
    document.getElementById(modalId).style.display = 'none';
    document.body.style.overflow = 'auto'; // Возврат прокрутки
}

// Закрытие при клике вне модального окна
window.onclick = function(event) {
    if (event.target.className === 'modal') {
        event.target.style.display = 'none';
        document.body.style.overflow = 'auto';
    }
}
